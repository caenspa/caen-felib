46class _Data:
47 """
48 Class representing data set by Node.set_read_data_format().
49 It holds a `numpy.ndarray` allocated with shape specified
50 in the data format.
51 """
52
53 # Public fields
54 name: str
55 type: str
56 dim: int
57 shape: List[int]
58 dtype: np.dtype
59 value: np.ndarray
60 proxy_value_2d: Optional[np.ndarray]
61 arg: Any
62
63 # Type aliases
64 class _DataField(TypedDict, total=False):
65 name: str
66 type: str
67 dim: int
68 shape: List[int]
69
70 def __init__(self, data_field: _DataField) -> None:
71
72 # Default attributes from fields passed to C library
73
74
75 self.namename = data_field['name']
76
77 self.typetype = data_field['type']
78
79 self.dimdim = data_field.get('dim', 0)
80
81 self.shapeshape = data_field.get('shape', [])
82
83 if self.dimdim != len(self.shapeshape):
84 raise RuntimeError('shape length must match dim')
85
86
87 self.dtypedtype = np.dtype(_type_map[self.typetype])
88
89 self.valuevalue = np.empty(self.shapeshape, dtype=self.dtypedtype)
90
91 self.proxy_value_2dproxy_value_2d = None
92
93 self.argarg = self.__generate_arg__generate_arg()
94
95 def __generate_arg(self) -> Any:
96 value: np.ndarray
97 if self.dimdim < 2:
98 # NumPy 0D and 1D arrays can be used directly.
99 value = self.valuevalue
100 else:
101 # NumPy 2D arrays cannot be directly used because they are
102 # implemented as contiguous memory blocks instead of arrays of
103 # pointers, used by CAEN_FELib.
104 # To overcome the problem we generate a proxy ndarray of pointers.
105 ptr_gen = (v.ctypes.data for v in self.valuevalue)
106 self.proxy_value_2dproxy_value_2d = np.fromiter(ptr_gen, dtype=ct.c_void_p)
107 value = self.proxy_value_2dproxy_value_2d
108 # value.ctypes is equivalent to value.ctypes.data_as(ctypes.c_void_p),
109 # that is fine for us.
110 return value.ctypes
111
112 def __repr__(self) -> str:
113 return f'{self.__class__.__name__}({self.name}, {self.type})'
114
115 def __str__(self) -> str:
116 return self.namename

373 def set_read_data_format(self, fmt: List[_Data._DataField]) -> Tuple[_Data, ...]:
374 """
375 Wrapper to CAEN_FELib_SetReadDataFormat()
376
377 In addition to what happens in C library, it also allocate data. Size of fields
378 with `dim > 0` must be specified by a `"shape"` entry in the field description,
379 that is a vector passed to the `shape` argument of `np.empty` constructor.
380 On fields with `dim == 0` the shape can be omitted, and is set to `[]` by default.
381 Fields can be accessed on data attribute of this class, that is a list of _Data
382 inizialized with the field descriptions, in the same order of @p format.
383
384 Example:
385 ```
386 nch = dig.par.numch.value
387 reclen = dig.par.recordlengths.value
388 format = [
389 {
390 'name': 'WAVEFORM',
391 'type': 'U16',
392 'dim': 2,
393 'shape': [nch, reclen],
394 },
395 ]
396 data = ep_node.set_read_data_format(format)
397 print(data[0])
398 ```
399
400 @param[in] fmt JSON representation of the format, in compliance with the endpoint "format" property (a list of dictionaries)
401 @return Tuple of _Data with allocated buffers of specified dim and shape, to be passed as second argument of read_data()
402 @exception error.Error in case of error
403 """
404 lib.set_read_data_format(self.handlehandle, dumps(fmt).encode())
405
406 # Important:
407 # Do not update lib.ReadData.argtypes with data.argtype because lib.ReadData
408 # is shared with all other endpoints and it would not be thread safe.
409 # More details on a comment on the caen_felib.lib constructor.
410 # Possible unsafe code could be something like:
411 # lib.ReadData.argtypes = [ct.c_uint64, ct.c_int] + [d.argtype for d in self.data]
412
413 # Allocate requested fields
414 return tuple(_Data(f) for f in fmt)

416 def read_data(self, timeout: int, data: Tuple[_Data, ...]) -> None:
417 """
418 Wrapper to CAEN_FELib_ReadData()
419
420 Unlike what happens in C library, variadic arguments are added automatically
421 according to what has been specified by set_read_data_format(). Data can be
422 retrieved using the data attribute of this class.
423
424 Example:
425 ```
426 # Start acquisition
427 dig.cmd.armacquisition()
428 dig.cmd.swstartacquisition()
429
430 while True:
431 try:
432 ep_node.read_data(100, data)
433 except error.Error as ex:
434 if ex.code == error.ErrorCode.Timeout:
435 continue
436 elif ex.code == error.ErrorCode.Stop:
437 break
438 else:
439 raise ex
440
441 # Do stuff with data
442 for field in data:
443 print(field.value)
444
445 dig.cmd.disarmacquisition()
446 ```
447
448 @param[in] timeout timeout of the function in milliseconds; if this value is -1 the function is blocking with infinite timeout
449 @param[out] data A tuple of _Data with requested data fields. The one returned by set_data_format() fits perfectly.
450 @exception error.Error in case of error
451 """
452 lib.read_data(self.handlehandle, timeout, *(d.arg for d in data))
