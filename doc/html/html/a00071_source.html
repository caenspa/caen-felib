20class CacheManager(List[_lru_cache_wrapper]):
21 """
22 A simple list of functions returned by `@lru_cache` decorator.
23
24 To be used with the optional parameter @p cache_manager of
25 lru_cache_method(), that will store a reference to the cached function
26 inside this list. This is a typing-safe way to call `cache_clear` and
27 `cache_info` of the internal cached functions, even if not exposed
28 directly by the inner function returned by lru_cache_method().
29 """
30 def clear_all(self) -> None:
31 """Invoke `cache_clear` on all functions in the list"""
32 for wrapper in self:
33 wrapper.cache_clear()

46 """
47 LRU cache decorator that keeps a weak reference to self.
48
49 To be used as decorator on methods that are known to return always
50 the same value. This can improve the performances of some methods
51 by a factor > 1000.
52 This wrapper using weak references is required: functools.lru_cache
53 holds a reference to all arguments: using directly on the methods it
54 would hold a reference to self, introducing subdle memory leaks.
55
56 @sa https://stackoverflow.com/a/68052994/3287591
57 """
58
59 def wrapper(method: Callable[Concatenate[_S, _P], _T]) -> Callable[Concatenate[_S, _P], _T]:
60
61 @lru_cache(maxsize, typed)
62 # ReferenceType is not subscriptable on Python <= 3.8
63 def cached_method(self_ref: ReferenceType, *args: _P.args, **kwargs: _P.kwargs) -> _T:
64 self = self_ref()
65 if self is not None:
66 return method(self, *args, **kwargs)
67 # self cannot be None: this function is always called by inner()
68 assert False, 'unreachable'
69
70 @wraps(method)
71 def inner(self: _S, *args: _P.args, **kwargs: _P.kwargs) -> _T:
72 # Ignore MyPy type checks because of bugs on lru_cache support.
73 # See https://stackoverflow.com/a/73517689/3287591.
74 return cached_method(ref(self), *args, **kwargs) # type: ignore
75
76 # Optionally store a reference to lru_cache decorated function to
77 # simplify cache management. See CacheManager documentation.
78 if cache_manager is not None:
79 cache_manager.append(cached_method)
80
81 return inner
82
83 return wrapper
